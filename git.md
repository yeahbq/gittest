cd = current dir
-mkdir = creates a dir
-touch = creates file 
-git init = create repo at location
-rm = remove
-ls -a = show hidden files 
-cd .. = back one dir
-cd ../.../ = more than one div
-cd or -cd~ = Home
-git add . = adds everything 
-git commit -m "..."
-git push origin master
-git log = Past commits
git status - tells changes to be committed
